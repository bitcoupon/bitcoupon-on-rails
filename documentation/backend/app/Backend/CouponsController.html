def index
  response.headers["token"] = "Your token: #{@public_key}"
  #binding.pry
  @coupons_all = Coupon.all
  @coupons = {
    pubkey: @public_key,
    coupons: @coupons_all
  }

  if @public_key.nil?
    render json: '{"error":"NO PUBLIC KEY PROVIDED"}', status: 401
  else
    render json: @coupons
  end
